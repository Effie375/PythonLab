```python
def readAndCheck():
  goon = True
  while goon:
    num = input("Δώσε αριθμό: ")
    while not num.isdigit():
      num = input("Δώσε αριθμό: ")
    num = int(num)
    if 0 <= num <= 10:
      goon = false
  return num
```

αριθμούς και θα την ταξινομεί.

```python
def sort(A):
  for i in range(1, len(A)):
    for j in range(len(A)-1, 0, -1):
      if A[j-1] > A[j]:
        temp = A[j-1]
        A[j-1] = A[j]
        A[j] = temp
  return A
```
